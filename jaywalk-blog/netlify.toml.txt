[build]
  publish = "."
  command = "echo 'No build command needed for static site'"

[[redirects]]
  from = "/admin/*"
  to = "/admin/index.html"
  status = 200

# Enable CORS for media
[[headers]]
  for = "/uploads/*"
  [headers.values]
    Access-Control-Allow-Origin = "*"

# Enable SPAs - redirect all 404s to index.html
[[redirects]]
  from = "/*"
  to = "/index.html"
  status = 200
  force = false
  conditions = {StatusCode = ["404"]}

# Security headers
[[headers]]
  for = "/*"
  [headers.values]
    X-Frame-Options = "DENY"
    X-XSS-Protection = "1; mode=block"
    X-Content-Type-Options = "nosniff"
